CREATE TABLE USER_INFO ( ID varchar(20) NOT NULL PRIMARY KEY, PWD varchar(20), NAME varchar(20) ,MAIN_EMOTION varchar(30));
CREATE TABLE USER_OWN ( ID varchar(20) NOT NULL, OWN_EMOTICON varchar(30),PRIMARY KEY(ID,OWN_EMOTICON));
CREATE TABLE EACH_EMOTICON ( IDX int(10) AUTO_INCREMENT PRIMARY KEY, EMOTICON_URL varchar(100) NOT NULL, EMOTICON_TYPE_NAME varchar(100) NOT NULL, EMOTION varchar(30) NOT NULL);
CREATE TABLE EMOTICON_TYPE_LINK ( EMOTICON_TYPE_NAME varchar(100) NOT NULL PRIMARY KEY, PURCHASE_LINK varchar(100) NOT NULL);
CREATE TABLE CHATTING ( IDX int(10) AUTO_INCREMENT PRIMARY KEY, CHAT_RECORD varchar(100) NOT NULL, EMOTION1 varchar(30) NOT NULL,EMOTION2 varchar(30),EMOTION3 varchar(30) );
CREATE TABLE EMOTION ( EMOTION varchar(30) NOT NULL PRIMARY KEY);
