CREATE TABLE USER_INFO ( ID varchar(10) primary key, PWD varchar(20), NAME varchar(20) ,MAIN_EMOTION varchar(20));
CREATE TABLE USER_OWN ( ID char(20) NOT NULL, OWN_EMOTICON varchar(30) NOT NULL, PRIMARY KEY(ID,OWN_EMOTICON));
CREATE TABLE EACH_EMOTICON ( EMOTICON_URL char(100) NOT NULL PRIMARY KEY, EMOTICON_TYPE_NAME char(100) NOT NULL, EMOTION char(30) NOT NULL);
CREATE TABLE EMOTICON_TYPE_LINK ( EMOTICON_TYPE_NAME char(100) NOT NULL PRIMARY KEY, PURCHASE_LINK char(100) NOT NULL);
CREATE TABLE CHATTING ( CHAT_RECORD char(100) NOT NULL PRIMARY KEY, EMOTION1 char(100) NOT NULL,EMOTION2 char(100),EMOTION3 char(100) );
