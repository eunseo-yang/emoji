CREATE TABLE USER_INFO ( IDX int(10) AUTO_ENCREMENT PRIMARY KEY, ID varchar(10) NOT NULL primary key, PWD varchar(20), NAME varchar(20) ,MAIN_EMOTION varchar(20));
CREATE TABLE USER_OWN ( IDX int(10) AUTO_ENCREMENT, ID char(20) NOT NULL, OWN_EMOTICON varchar(30), PRIMARY KEY(IDX,ID));
CREATE TABLE EACH_EMOTICON ( IDX int(10) AUTO_ENCREMENT PRIMARY KEY, EMOTICON_URL char(100) NOT NULL, EMOTICON_TYPE_NAME char(100) NOT NULL, EMOTION char(30) NOT NULL);
CREATE TABLE EMOTICON_TYPE_LINK ( IDX int(10) AUTO_ENCREMENT, EMOTICON_TYPE_NAME char(100) NOT NULL, PURCHASE_LINK char(100) NOT NULL, PRIMARY KEY(IDX,EMOTICON_TYPE_NAME));
CREATE TABLE CHATTING ( IDX int(10) AUTO_ENCREMENT, CHAT_RECORD char(100) NOT NULL, EMOTION1 char(100) NOT NULL,EMOTION2 char(100),EMOTION3 char(100),PRIMARY KEY(IDX,CHAT_RECORD,EMOTION1) );
